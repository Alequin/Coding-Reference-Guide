Structure

A structure is a value data type which allows for the creation of simple Objects which may hold multiple properties. They can be useful as they require less memory that an object created from a class.

Stucts should be defined either within the namespace or class blocks but outside any method blocks. They cannot inherit functionality from other structs or classes but may implement interfaces.

Below are a few examples of how structs may be used.

struct Building{

  public String name;
  public uint floors;

  public Building(String name, uint floors){
    this.name = name;
    this.floors = floors;
  }
};

Building skyScraper = new Building("Gurking", 150);
Console.WriteLine(skyScraper.name+" "+skyScraper.floors);
//=> Gurking 150

struct Animal{
  public String name;
  public uint age;
  public String type;

  public static Animal dog(String name){
    Animal dog;
    dog.name = name;
    dog.age = 0;
    dog.type = "dog";
    return dog;
  }

  public void logAnimal(){
    Console.WriteLine(this.name + " " + this.age + " " + this.type);
  }
};

Animal cat;

cat.name = "Billy";
cat.age = 4;
cat.type = "cat";
Console.WriteLine(cat.name +" "+ cat.age +" "+ cat.type);
cat.logAnimal();

Animal dog = Animal.dog("Teddy");

dog.logAnimal();
//=> Billy 4 cat
Billy 4 cat Teddy 0 dog
